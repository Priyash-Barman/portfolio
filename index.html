@Injectable()
export class AppService {
  constructor(private readonly prisma: PrismaService) {}

  async getDashboardData(userId: string) {
    const user = await this.prisma.user.findUnique({
      where: { id: userId },
      include: { projects: true }
    });
    return {
      status: 'active',
      data: user
    };
  }
}
